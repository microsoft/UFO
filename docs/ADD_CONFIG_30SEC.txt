────────────────────────────────────────────────────────────────────

    # config/ufo/agents.yaml（编辑现有文件或新建）
    
    EVALUATOR_AGENT:
      API_TYPE: "openai"
      API_MODEL: "gpt-4"
      API_KEY: "YOUR_KEY"
      MAX_TOKENS: 2000
      TEMPERATURE: 0.7


💻 Python 代码
────────────────────────────────────────────────────────────────────

    config = get_ufo_config()
    
    # ✅ 立即可用！无需修改任何其他代码

  访问方式对比
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

    YAML 配置：
    ──────────────────────────────────────
    NEW_SETTING: 123
    
    NESTED:
      value: "abc"
    
    
    Python 访问：
    ──────────────────────────────────────
    # 方式 1: 属性访问（推荐，有 IDE 补全）
    config.NEW_SETTING          → 123
    config.NESTED.value         → "abc"
    
    # 方式 2: 字典访问（兼容旧代码）
    config["NEW_SETTING"]       → 123
    config["NESTED"]["value"]   → "abc"

    ──────────────────────────────────────

    ❸ 添加列表配置
    ──────────────────────────────────────

              host: localhost
              port: 5432
